\subsection{Tratamento da Imagem}
\setbeamercovered{transparent}
\begin{frame}\frametitle{Tratamento da Imagem}
	% Falar sobre o que podemos fazer para melhorar a imagem 
	\begin{itemize}
	\item Limiarização usada Threshold de Otsu: Maximiza a variação intraclasse.

	\item Laplaciano: Derivada segunda.
		
	\item Transformada Distância Euclidiana: Distância entre o pixel e o primeiro ponto preto $dist(p,q) = \sqrt{(p_1 - q_1)^2 + (p_2 - q_2)^2 + ... + (p_n - q_n)^2}$.
	
	\item Limiarização - Coeficiente escolhido.
	
	\item Dilatação
	
	\item Cria o Marcador para a segmentação de Watershed.
	
	\end{itemize}	
	
\end{frame}
